Goal was to reproduce a published tsne plot from Kakaradov et al.

Starting from a tabular file with TPM values, the authors suggest filtering by: We detected over 6,000 genes with a mean expression of at least 1 transcript per million reads (TPM) per cell. Test text

library(Rtsne)
count<- read.csv("GSE89405_Kakaradov_singlecell_expression_matrix.csv", 
                 header = TRUE, row.names = 2)
#filter out useless column 
count_wo_x <- subset(count, select=-c(X))
#get rid of missing values, I believe that one row (one cell) is full of NAs
count_wo_x <- count_wo_x[complete.cases(count_wo_x),]
#GSM number : cell ID dictionary
gsm_cell <- read.csv("gsm_cell.txt", sep = '\t', header = FALSE)
#meta data table
meta_table <- read.csv("SraRunTable.txt", sep = '\t', header = TRUE)
#merge the two
new_meta_table <- merge(meta_table, gsm_cell, by.x = 'Sample_Name', by.y = 'V1')
#subset the merged filed to generate a new cell subtype (T cell subset)
# to cell ID dictionary 
new_dict <- subset(new_meta_table, select=c(subtype, V2))
#make column names more informative
colnames(new_dict) <- c("subtype", "cell_id")
#match the order of the dictionary with the rows in the expression matrix
new_dict <- new_dict[match(row.names(count_wo_x), new_dict$cell_id),]
#leftover factors need to be removed
new_dict$subtype <- factor(new_dict$subtype)
#I believe that matrices are more memory efficient than data frames
mat_count_wo_x <- as.matrix(count_wo_x)
#filter based on instruction from paper, 6,000 genes with in average 1 TPM/cell
TPM_above_1 = apply(mat_count_wo_x, 2, sum)
division = TPM_above_1/287
head(division) 
##      Gnai3       Pbsn      Cdc45      Scml2       Apoh       Narf 
## 1.96972234 0.00000000 1.78248246 0.03625264 0.02226975 0.52694757
filter <- as.logical(division >= 1)
head(TPM_above_1)
##      Gnai3       Pbsn      Cdc45      Scml2       Apoh       Narf 
## 565.310313   0.000000 511.572467  10.404507   6.391418 151.233954
matrix_filtered <- mat_count_wo_x[,filter]
#set seed
set.seed(100)
#run tsne
tsne_out_filtered <- Rtsne(matrix_filtered,pca=FALSE,perplexity=30,theta=0.0) # Run TSNE
#transfer tsne data to a dataframe for easy plotting
tsne_plot <- data.frame(x = tsne_out_filtered$Y[,1], y = tsne_out_filtered$Y[,2], col = new_dict$subtype)
